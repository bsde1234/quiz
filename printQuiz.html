Finishing Steps

  1. Copy the below code.
  2. Paste the code into a notepad or similar document.
  3. Save the file as a .html extension. Example: quiz.html.
  4. Host your quiz on your server. Alternatively, if you do not have access to a web server, you can try opening your saved file in your browser. It will load as a local web page.

<script>

const questions = {{questionsKey}};
const answerList = {{storeAnswersKey}};
const key = {{answersKey}};
const userKey = [];

function loadQuiz () {
for (i = 0; i < answerList.length; i++) {
document.getElementById("quizArea").innerHTML +=
"<div class = 'test'>" + (i+1) + ". " + questions[i] + "</div>" +
"<div class = 'answerArea'>" +
"<input type = 'radio' class = 'answers' + name ='" + i + "'value = 0>" + "</input>" +
answerList[i][0] + "</div>";
if (typeof answerList[i][1] === "undefined") {
} else {
document.getElementById("quizArea").innerHTML +=
"<div class = 'answerArea'>" +
"<input type = 'radio' class = 'answers' name ='" + i + "'value = 1>" + "</input>" +
answerList[i][1] + "</div>";
if (typeof answerList[i][2] === "undefined") {
} else {
document.getElementById("quizArea").innerHTML += "<div class = 'answerArea'>" +
"<input type = 'radio' class = 'answers' name ='" + i + "'value = 2>" + "</input>" +
answerList[i][2] + "</div>";
if (typeof answerList[i][3] === "undefined") {
} else {
document.getElementById("quizArea").innerHTML +=
"<div class = 'answerArea'>" +
"<input type = 'radio' class = 'answers' name ='" + i + "'value = 3>" + "</input>" +
answerList[i][3] + "</div>";
}
}
}
}
document.getElementById("quizArea").innerHTML +=
"<div class = 'score' onclick = 'score()' > Score </div>" +
"<div id = 'scoreboard'></div>";
}

function score () {
var correct = 0;
var answerLen = 0;
var gatherAnswers = document.getElementsByClassName('answerArea');
var gatherQuestions = document.getElementsByClassName('test');
var gatherQuest = document.getElementsByClassName('answers');
for (i = 0; i < gatherQuest.length; i++) {
if (gatherQuest[i].checked) {
userKey.push(gatherQuest[i].value);
}
}
for (j = 0; j < key.length; j++) {
var ans = answerLen + parseInt(key[j]);
if (key[j] === userKey[j]) {
correct += 1;
gatherQuestions[j].style.backgroundColor = '#006600';
gatherQuestions[j].style.color = 'white';
}
if (key[j] !== userKey[j]) {
gatherAnswers[ans].style.backgroundColor = "#006600";
gatherAnswers[ans].style.color = 'white';
gatherQuestions[j].style.backgroundColor = '#990000';
gatherQuestions[j].style.color = 'white';
gatherQuestions[j].innerHTML += "</br>" + "The correct answer was choice # " +
(parseInt(key[j]) + 1);
}
answerLen += answerList[j].length;
}
document.getElementById("scoreboard").innerHTML =
Math.floor(correct/key.length * 100) + ". You got " +
correct + " out of " + key.length + " correct.";
document.getElementsByClassName('score')[0].style.visibility = 'hidden';
}
